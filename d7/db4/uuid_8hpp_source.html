21 void inline m128itos(__m128i x, char *mem)
22 {
23 // Expand each byte in x to two bytes in res
24 // i.e. 0x12345678 -> 0x0102030405060708
25 // Then translate each byte to its hex ascii representation
26 // i.e. 0x0102030405060708 -> 0x3132333435363738
27 const __m256i mask = _mm256_set1_epi8(0x0F);
28 const __m256i add = _mm256_set1_epi8(0x06);
29 const __m256i alpha_mask = _mm256_set1_epi8(0x10);
30 const __m256i alpha_offset = _mm256_set1_epi8(0x57);
31
32 __m256i a = _mm256_castsi128_si256(x);
33 __m256i as = _mm256_srli_epi64(a, 4);
34 __m256i lo = _mm256_unpacklo_epi8(as, a);
35 __m128i hi = _mm256_castsi256_si128(_mm256_unpackhi_epi8(as, a));
36 __m256i c = _mm256_inserti128_si256(lo, hi, 1);
37 __m256i d = _mm256_and_si256(c, mask);
38 __m256i alpha = _mm256_slli_epi64(_mm256_and_si256(_mm256_add_epi8(d, add), alpha_mask), 3);
39 __m256i offset = _mm256_blendv_epi8(_mm256_slli_epi64(add, 3), alpha_offset, alpha);
40 __m256i res = _mm256_add_epi8(d, offset);
41
42 // Add dashes between blocks as specified in RFC-4122
43 // 8-4-4-4-12
44 const __m256i dash_shuffle =
45 _mm256_set_epi32(0x0b0a0908, 0x07060504, 0x80030201, 0x00808080, 0x0d0c800b, 0x0a090880,
46 0x07060504, 0x03020100);
47 const __m256i dash = _mm256_set_epi64x(0x0000000000000000ull, 0x2d000000002d0000ull,
48 0x00002d000000002d, 0x0000000000000000ull);
49
50 __m256i resd = _mm256_shuffle_epi8(res, dash_shuffle);
51 resd = _mm256_or_si256(resd, dash);
52
53 _mm256_storeu_si256((__m256i *)mem, endianness::betole256(resd));
54 *(uint16_t *)(mem + 16) = endianness::betole16(_mm256_extract_epi16(res, 7));
55 *(uint32_t *)(mem + 32) = endianness::betole32(_mm256_extract_epi32(res, 7));
56 }

65 __m128i inline stom128i(const char *mem)
66 {
67 // Remove dashes and pack hex ascii bytes in a 256-bits int
68 const __m256i dash_shuffle =
69 _mm256_set_epi32(0x80808080, 0x0f0e0d0c, 0x0b0a0908, 0x06050403, 0x80800f0e, 0x0c0b0a09,
70 0x07060504, 0x03020100);
71
72 __m256i x = endianness::betole256(_mm256_loadu_si256((__m256i *)mem));
73 x = _mm256_shuffle_epi8(x, dash_shuffle);
74 x = _mm256_insert_epi16(x, endianness::betole16(*(uint16_t *)(mem + 16)), 7);
75 x = _mm256_insert_epi32(x, endianness::betole32(*(uint32_t *)(mem + 32)), 7);
76
77 // Build a mask to apply a different offset to alphas and digits
78 const __m256i sub = _mm256_set1_epi8(0x2F);
79 const __m256i mask = _mm256_set1_epi8(0x20);
80 const __m256i alpha_offset = _mm256_set1_epi8(0x28);
81 const __m256i digits_offset = _mm256_set1_epi8(0x01);
82 const __m256i unweave = _mm256_set_epi32(0x0f0d0b09, 0x0e0c0a08, 0x07050301, 0x06040200,
83 0x0f0d0b09, 0x0e0c0a08, 0x07050301, 0x06040200);
84 const __m256i shift = _mm256_set_epi32(0x00000000, 0x00000004, 0x00000000, 0x00000004,
85 0x00000000, 0x00000004, 0x00000000, 0x00000004);
86
87 // Translate ascii bytes to their value
88 // i.e. 0x3132333435363738 -> 0x0102030405060708
89 // Shift hi-digits
90 // i.e. 0x0102030405060708 -> 0x1002300450067008
91 // Horizontal add
92 // i.e. 0x1002300450067008 -> 0x12345678
93 __m256i a = _mm256_sub_epi8(x, sub);
94 __m256i alpha = _mm256_slli_epi64(_mm256_and_si256(a, mask), 2);
95 __m256i sub_mask = _mm256_blendv_epi8(digits_offset, alpha_offset, alpha);
96 a = _mm256_sub_epi8(a, sub_mask);
97 a = _mm256_shuffle_epi8(a, unweave);
98 a = _mm256_sllv_epi32(a, shift);
99 a = _mm256_hadd_epi32(a, _mm256_setzero_si256());
100 a = _mm256_permute4x64_epi64(a, 0b00001000);
101
102 return _mm256_castsi256_si128(a);
103 }

308 UUID getUUID()
309 {
310 // The two masks set the uuid version (4) and variant (1)
311 const __m128i and_mask = _mm_set_epi64x(0xFFFFFFFFFFFFFF3Full, 0xFF0FFFFFFFFFFFFFull);
312 const __m128i or_mask = _mm_set_epi64x(0x0000000000000080ull, 0x0040000000000000ull);
313 __m128i n = _mm_set_epi64x(distribution(*generator), distribution(*generator));
314 __m128i uuid = _mm_or_si128(_mm_and_si128(n, and_mask), or_mask);
315
316 return UUID(uuid);
317 }
