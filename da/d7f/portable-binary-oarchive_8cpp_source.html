19 void portable_binary_oarchive::save_impl(const boost::intmax_t l, [[maybe_unused]] const char maxsize)
20 {
21 char size = 0;
22
23 if (l == 0)
24 {
25 this->primitive_base_t::save(size);
26 return;
27 }
28
29 boost::intmax_t ll;
30 bool negative = (l < 0);
31 if (negative)
32 ll = -l;
33 else
34 ll = l;
35
36 do
37 {
38 ll >>= CHAR_BIT;
39 ++size;
40 } while (ll != 0);
41
42 this->primitive_base_t::save(static_cast<char>(negative ? -size : size));
43
44 if (negative)
45 ll = -l;
46 else
47 ll = l;
48 char *cptr = reinterpret_cast<char *>(&ll);
49#if BOOST_ENDIAN_BIG_BYTE
50 cptr += (sizeof(boost::intmax_t) - size);
51 if (m_flags & endian_little)
52 reverse_bytes(size, cptr);
53#else
54 if (m_flags & endian_big)
55 reverse_bytes(size, cptr);
56#endif
57 this->primitive_base_t::save_binary(cptr, size);
58 }

60 void portable_binary_oarchive::init(unsigned int flags)
61 {
62 if (m_flags == (endian_big | endian_little))
63 {
64 boost::serialization::throw_exception(portable_binary_oarchive_exception());
65 }
66 if (0 == (flags & ::boost::archive::no_header))
67 {
68 // write signature in an archive version independent manner
69 const std::string file_signature(::boost::archive::BOOST_ARCHIVE_SIGNATURE());
70 *this << file_signature;
71 // write library version
72 const ::boost::archive::library_version_type v(::boost::archive::BOOST_ARCHIVE_VERSION());
73 *this << v;
74 }
75 save(static_cast<unsigned char>(m_flags >> CHAR_BIT));
76 }
