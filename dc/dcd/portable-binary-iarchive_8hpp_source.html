38 // exception to be thrown if integer read from archive doesn't fit
39 // variable being loaded
40 class portable_binary_iarchive_exception : public ::boost::archive::archive_exception
41 {
42 public:
43 enum exception_code
44 {
45 incompatible_integer_size
46 } m_exception_code;
47 portable_binary_iarchive_exception(exception_code c = incompatible_integer_size)
48 : ::boost::archive::archive_exception(::boost::archive::archive_exception::other_exception), m_exception_code(c)
49 {
50 }
51 virtual const char *what() const throw()
52 {
53 const char *msg = "programmer error";
54 switch (m_exception_code)
55 {
56 case incompatible_integer_size:
57 msg = "integer cannot be represented";
58 break;
59 default:
60 msg = ::boost::archive::archive_exception::what();
61 assert(false);
62 break;
63 }
64 return msg;
65 }
