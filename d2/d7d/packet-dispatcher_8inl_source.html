166 inline bool PacketDispatcher::fulfill_promises(UniquePacketID packet_id, BasePacketPtr &packet)
167 {
168 // Fulfill first filtered promise in filter_promise_map
169 {
170 auto it = promise_filter_map_.find(packet_id);
171 if (it != promise_filter_map_.end())
172 {
173 for (auto &promise_filter : it->second)
174 {
175 if (!promise_filter.first || promise_filter.first(packet))
176 {
177 promise_filter.second->set_value(std::move(packet));
178 spdlog::trace("Fulfilled filtered promise for packet_id: {}", packet_id);
179 return true;
180 }
181 }
182 }
183 }
184
185 // Fulfill the first promise in promise_map
186 {
187 auto it = promise_map_.find(packet_id);
188 if (it != promise_map_.end() && !it->second.empty())
189 {
190 it->second.front()->set_value(std::move(packet));
191 it->second.pop_front();
192 spdlog::trace("Fulfilled promise for packet_id: {}", packet_id);
193 return true;
194 }
195 }
196
197 spdlog::trace("No promises to fulfill for packet_id: {}", packet_id);
198 return false;

201 inline bool PacketDispatcher::fulfill_handlers(UniquePacketID packet_id, BasePacketPtr &packet,
202 float &min_handler_timestamp,
203 utils::SteadyTimer &timer)
204 {
205 auto it = default_handlers_.find(packet_id);
206 if (it == default_handlers_.end())
207 {
208 spdlog::warn("No handlers to fulfill for packet_id: {}", packet_id);
209 return false;
210 }
211
212 for (auto &[delay, filter, handler] : it->second)
213 {
214 if (delay > packet->get_packet_time_alive())
215 {
216 min_handler_timestamp =
217 std::min<float>(min_handler_timestamp,
218 timer.elapsed() + delay - packet->get_packet_time_alive());
219 spdlog::trace("Handler delay for packet_id {} is greater than packet time alive.",
220 packet_id);
221 continue;
222 }
223
224 if (bool(filter) && !filter(*packet))
225 {
226 spdlog::trace("Filter condition not satisfied for packet_id: {}", packet_id);
227 continue;
228 }
229
230 handler(std::move(packet));
231 spdlog::trace("Fulfilled handler for packet_id: {}", packet_id);
232 return true;
233 }
234
235 spdlog::trace("No suitable handlers to fulfill for packet_id: {}", packet_id);
236 return false;
