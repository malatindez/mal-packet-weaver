46 boost::asio::awaitable<void> PacketDispatcher::Run()
47 {
48 ExponentialBackoff backoff{ std::chrono::microseconds(1), std::chrono::microseconds(100), 2, 32, 0.1 };
49 SteadyTimer timer;
50 float min_handler_timestamp = std::numeric_limits<float>::max();
51
52 std::shared_ptr<PacketDispatcher> dispatcher_lock = co_await get_shared_ptr();
53 if (dispatcher_lock == nullptr)
54 {
55 spdlog::error(
56 "Couldn't retrieve shared pointer for session. Did you create the "
57 "session using std::make_shared?");
58 co_return;
59 }
60 try
61 {
62 while (alive_.load())
63 {
64 bool updated = co_await pop_inputs();
65
66 if (!updated)
67 {
68 if (min_handler_timestamp < timer.elapsed())
69 {
70 spdlog::trace("Updating handlers...");
71 min_handler_timestamp = std::numeric_limits<float>::max();
72
73 for (auto &[packet_id, packet_vector] : unprocessed_packets_)
74 {
75 // Remove packets that fulfill handlers and update min_handler_timestamp
76 std::erase_if(
77 packet_vector, [this, &packet_id, &min_handler_timestamp, &timer](BasePacketPtr &packet)
78 __lambda_force_inline
79 { return fulfill_handlers(packet_id, packet, min_handler_timestamp, timer); });
80 }
81 }
82
83 // Introduce delay and increase it using exponential backoff strategy
84 boost::asio::steady_timer async_timer(co_await boost::asio::this_coro::executor,
85 backoff.get_current_delay());
86 co_await async_timer.async_wait(boost::asio::use_awaitable);
87 backoff.increase_delay();
88 continue;
89 }
90
91 spdlog::trace("Input arrays were updated! Fetching...");
92
93 min_handler_timestamp = std::numeric_limits<float>::max();
94 for (auto &[packet_id, packet_vector] : unprocessed_packets_)
95 {
96 // Process packets: fulfill promises, fulfill handlers, and check for expiration
97 std::erase_if(packet_vector,
98 [this, &packet_id, &min_handler_timestamp, &timer](BasePacketPtr &packet)
99 __lambda_force_inline
100 {
101 return fulfill_promises(packet_id, packet) ||
102 fulfill_handlers(packet_id, packet, min_handler_timestamp, timer) ||
103 packet->expired();
104 });
105 }
106
107 // Remove empty entries from the unprocessed_packets_ map
108 std::erase_if(unprocessed_packets_,
109 [](auto const &pair) __lambda_force_inline { return pair.second.empty(); });
110
111 // Decrease the delay for exponential backoff
112 backoff.decrease_delay();
113 }
114 }
115 catch (std::exception &e)
116 {
117 spdlog::error("PacketDispatcher::Run cancelled with an error: {}", e.what());
118 }
119 spdlog::info("Exiting PacketDispatcher::Run");
120 }
